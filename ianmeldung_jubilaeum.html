    <script type="module">

        // Firebase SDKs importieren

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getAuth, signInAnonymously } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";

        import { getFirestore, collection, addDoc, onSnapshot, query } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";


        // --- Globale Variablen (werden von der Umgebung bereitgestellt) ---

        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : { apiKey: "...", authDomain: "...", projectId: "..." };

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-jubilaeum-app';


        // --- Firebase initialisieren ---

        const app = initializeApp(firebaseConfig);

        const auth = getAuth(app);

        const db = getFirestore(app);


        // --- DOM-Elemente abrufen ---

        const anmeldeFormular = document.getElementById('anmelde-formular');

        const detailsContainer = document.getElementById('details-container');

        const begleitungDetails = document.getElementById('begleitung-details');

        const dessertDetails = document.getElementById('dessert-details');

        const dessertListeContainer = document.getElementById('dessert-liste-container');

        const dessertListe = document.getElementById('dessert-liste');

        const anmeldungContainer = document.getElementById('anmeldung-container');

        const erfolgsmeldung = document.getElementById('erfolgsmeldung');

        

        // --- Hilfsfunktion für Animationen ---

        function toggleSection(element, show) {

            if (show) {

                element.classList.remove('hidden');

                setTimeout(() => {

                    element.classList.remove('opacity-0', 'max-h-0');

                    element.style.maxHeight = element.scrollHeight + 'px';

                }, 10);

            } else {

                element.classList.add('opacity-0');

                element.style.maxHeight = '0px';

                 setTimeout(() => {

                    if (element.style.maxHeight === '0px') {

                         element.classList.add('hidden');

                    }

                }, 500);

            }

        }


        // --- Event Listener für bedingte Logik (UI-Animationen) ---

        anmeldeFormular.elements.teilnahme.forEach(radio => {

            radio.addEventListener('change', (e) => {

                const showDetails = e.target.value === 'ja';

                toggleSection(detailsContainer, showDetails);

                if(showDetails) dessertListeContainer.classList.remove('hidden');

                else dessertListeContainer.classList.add('hidden');

            });

        });


        anmeldeFormular.elements.begleitung.forEach(radio => {

            radio.addEventListener('change', (e) => {

                toggleSection(begleitungDetails, e.target.value === 'ja');

            });

        });


        anmeldeFormular.elements['dessert-beitrag'].forEach(radio => {

            radio.addEventListener('change', (e) => {

                // BUGFIX: Variable war falsch geschrieben (dessert-details vs dessertDetails)

                toggleSection(dessertDetails, e.target.value === 'ja');

            });

        });


        // --- Hauptfunktion der Anwendung ---

        async function main() {

            try {

                // PERMISSION FIX: Zuerst anonym anmelden und warten, bevor Datenbankzugriffe erlaubt werden.

                await signInAnonymously(auth);

                console.log("Anonyme Anmeldung erfolgreich. Firestore-Listener werden jetzt eingerichtet.");


                // --- Listener für Formular-Einreichung ---

                anmeldeFormular.addEventListener('submit', async (e) => {

                    e.preventDefault();

                    const formData = new FormData(anmeldeFormular);

                    const data = Object.fromEntries(formData.entries());


                    try {

                        await addDoc(collection(db, `/artifacts/${appId}/public/data/anmeldungen`), { ...data, timestamp: new Date() });


                        if (data.teilnahme === 'ja' && data['dessert-beitrag'] === 'ja' && data['dessert-was']) {

                            await addDoc(collection(db, `/artifacts/${appId}/public/data/desserts`), {

                                name: data.name,

                                dessert: data['dessert-was'],

                                timestamp: new Date()

                            });

                        }

                        

                        anmeldungContainer.classList.add('hidden');

                        dessertListeContainer.classList.add('hidden'); // Dessert-Liste auch ausblenden

                        erfolgsmeldung.classList.remove('hidden');


                    } catch (error) {

                        console.error("Fehler beim Speichern der Anmeldung: ", error);

                        alert("Es ist ein Fehler aufgetreten. Bitte versuche es erneut.");

                    }

                });


                // --- Live-Listener für Dessert-Liste ---

                const dessertsRef = collection(db, `/artifacts/${appId}/public/data/desserts`);

                const q = query(dessertsRef);


                onSnapshot(q, (querySnapshot) => {

                    if (querySnapshot.empty) {

                        dessertListe.innerHTML = '<p class="text-center text-gray-500">Noch keine Desserts eingetragen. Sei der/die Erste!</p>';

                        return;

                    }

                    

                    dessertListe.innerHTML = '';

                    querySnapshot.forEach((doc) => {

                        const dessertData = doc.data();

                        const eintrag = document.createElement('div');

                        eintrag.className = 'flex justify-between items-center bg-gray-50 p-3 rounded-lg border';

                        eintrag.innerHTML = `

                            <span class="font-medium text-gray-800">${dessertData.dessert}</span>

                            <span class="text-sm text-gray-500">von ${dessertData.name}</span>

                        `;

                        dessertListe.appendChild(eintrag);

                    });

                }, (error) => {

                    console.error("Fehler beim Abrufen der Dessert-Liste:", error);

                    dessertListe.innerHTML = '<p class="text-center text-red-500">Fehler beim Laden der Liste.</p>';

                });


            } catch (error) {

                console.error("Kritischer Fehler bei der App-Initialisierung:", error);

                // Hier könnte man eine Fehlermeldung für den Benutzer anzeigen

            }

        }

        

        // App starten

        main();


    </script>
